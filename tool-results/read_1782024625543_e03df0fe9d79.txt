   104→   104→/** Convert an API message (loaded from DB) to our ChatMessageData shape. */
   105→   105→function apiMessageToChatMessage(
   106→   106→  m: ApiMessage,
   107→   107→  agentType?: string,
   108→   108→  agentName?: string,
   109→   109→): ChatMessageData {
   110→   110→  const role: ChatMessageData['role'] =
   111→   111→    m.role === 'user' ? 'user' :
   112→   112→    m.role === 'assistant' ? 'assistant' :
   113→   113→    m.role === 'system' ? 'system' : 'system';
   114→   114→  const tokensUsed = (m.tokensInput || 0) + (m.tokensOutput || 0);
   115→   115→  return {
   116→   116→    id: m.id,
   117→   117→    role,
   118→   118→    content: m.content || '',
   119→   119→    createdAt: m.createdAt,
   120→   120→    agentType,
   121→   121→    agentName,
   122→   122→    isStreaming: false,
   123→   123→    metadata: role === 'assistant' && (tokensUsed || m.cost || m.latencyMs) ? {
   124→   124→      tokensUsed: tokensUsed || undefined,
   125→   125→      cost: m.cost ? parseFloat(m.cost) : undefined,
   126→   126→      durationMs: m.latencyMs,
   127→   127→    } : undefined,
   128→   128→  };
   129→   129→}

   131→   131→function formatTotalCost(s: string | number): string {
   132→   132→  const n = typeof s === 'number' ? s : parseFloat(s);
   133→   133→  if (!isFinite(n) || n === 0) return '$0.00';
   134→   134→  if (n < 0.01) return `$${n.toFixed(5)}`;
   135→   135→  return `$${n.toFixed(4)}`;
   136→   136→}
   137→   137→
   138→   138→function formatTokens(n: number): string {
   139→   139→  if (n >= 1_000_000) return `${(n / 1_000_000).toFixed(1)}M`;
   140→   140→  if (n >= 1_000) return `${(n / 1_000).toFixed(1)}k`;
   141→   141→  return String(n);
   142→   142→}

   456→   456→      const finishRunningSteps = (steps: ThinkingStep[]): ThinkingStep[] =>
   457→   457→        steps.map(s => s.status === 'running' ? { ...s, status: 'completed' as const } : s);
   458→   458→
   459→   459→      while (true) {
   460→   460→        const { done, value } = await reader.read();
   461→   461→        if (done) break;
   462→   462→        buffer += decoder.decode(value, { stream: true });
   463→   463→
   464→   464→        // SSE messages are separated by double newlines; lines start with "data: "
   465→   465→        const lines = buffer.split('\n');
   466→   466→        buffer = lines.pop() || ''; // keep last partial line
   467→   467→
   468→   468→        for (const rawLine of lines) {
   469→   469→          const line = rawLine.trim();
   470→   470→          if (!line.startsWith('data:')) continue;
   471→   471→          const payload = line.slice(5).trim();
   472→   472→          if (payload === '[DONE]') continue;
   473→   473→          if (!payload) continue;
   474→   474→
   475→   475→          let event: SSEEvent;
   476→   476→          try {
   477→   477→            event = JSON.parse(payload);
   478→   478→          } catch {
   479→   479→            continue;
   480→   480→          }
   481→   481→
   482→   482→          switch (event.type) {
   483→   483→            case 'started': {
   484→   484→              startedAt = Date.now();
   485→   485→              updateAssistant(m => ({
   486→   486→                ...m,
   487→   487→                thinkingSteps: m.thinkingSteps?.map(s =>
   488→   488→                  s.kind === 'analyzing' && s.status === 'running'
   489→   489→                    ? { ...s, status: 'running', label: 'Analyzing request' }
   490→   490→                    : s
   491→   491→                ) ?? [],
   492→   492→              }));
   493→   493→              break;
   494→   494→            }
   495→   495→            case 'thinking': {
   496→   496→              updateAssistant(m => ({
   497→   497→                ...m,
   498→   498→                thinkingSteps: [
   499→   499→                  ...(m.thinkingSteps ?? []),
   500→   500→                  {
   501→   501→                    id: newStepId(),
   502→   502→                    kind: 'info',
   503→   503→                    label: 'Thinking',
   504→   504→                    detail: event.content.slice(0, 100),
   505→   505→                    status: 'completed',
   506→   506→                    ts: Date.now(),
   507→   507→                  },
   508→   508→                ],
   509→   509→              }));
   510→   510→              break;
   511→   511→            }
   512→   512→            case 'message_chunk': {
   513→   513→              accumulatedContent += event.content;
   514→   514→              updateAssistant(m => {
   515→   515→                let steps = m.thinkingSteps ?? [];
   516→   516→                // Complete any 'analyzing' step, ensure a 'synthesizing' step exists
   517→   517→                steps = steps.map(s => s.status === 'running' && s.kind === 'analyzing'
   518→   518→                  ? { ...s, status: 'completed' as const, durationMs: Date.now() - startedAt }
   519→   519→                  : s
   520→   520→                );
   521→   521→                if (!steps.some(s => s.kind === 'synthesizing')) {
   522→   522→                  steps = [...steps, {
   523→   523→                    id: newStepId(), kind: 'synthesizing',
   524→   524→                    label: 'Synthesizing response',
   525→   525→                    status: 'running', ts: Date.now(),
   526→   526→                  }];
   527→   527→                }
   528→   528→                return { ...m, content: accumulatedContent, thinkingSteps: steps };
   529→   529→              });
   530→   530→              break;
   531→   531→            }
   532→   532→            case 'tool_call': {
   533→   533→              updateAssistant(m => {
   534→   534→                let steps = m.thinkingSteps ?? [];
   535→   535→                // Complete any running 'synthesizing' step
   536→   536→                steps = steps.map(s => s.status === 'running' && s.kind === 'synthesizing'
   537→   537→                  ? { ...s, status: 'completed' as const }
   538→   538→                  : s
   539→   539→                );
   540→   540→                // Add a 'tool' thinking step
   541→   541→                steps = [...steps, {
   542→   542→                  id: newStepId(),
   543→   543→                  kind: 'tool',
   544→   544→                  label: `Running ${event.toolName}`,
   545→   545→                  status: 'running',
   546→   546→                  ts: Date.now(),
   547→   547→                }];
   548→   548→                // Add tool call card (running)
   549→   549→                const newTool: ToolCallData = {
   550→   550→                  id: event.toolCallId || `tc-${Date.now()}-${Math.random().toString(36).slice(2, 6)}`,
   551→   551→                  toolName: event.toolName,
   552→   552→                  args: event.args,
   553→   553→                  status: 'running',
   554→   554→                };
   555→   555→                return {
   556→   556→                  ...m,
   557→   557→                  thinkingSteps: steps,
   558→   558→                  toolCalls: [...(m.toolCalls ?? []), newTool],
   559→   559→                };
   560→   560→              });
   561→   561→              break;
   562→   562→            }
   563→   563→            case 'tool_result': {
   564→   564→              updateAssistant(m => {
   565→   565→                let steps = m.thinkingSteps ?? [];
   566→   566→                // Complete the matching 'tool' step
   567→   567→                steps = steps.map(s => {
   568→   568→                  if (s.status === 'running' && s.kind === 'tool' &&
   569→   569→                      s.label === `Running ${event.toolName}`) {
   570→   570→                    return { ...s, status: 'completed' as const, durationMs: event.durationMs };
   571→   571→                  }
   572→   572→                  return s;
   573→   573→                });
   574→   574→                // Re-open 'synthesizing' step if none running
   575→   575→                if (!steps.some(s => s.status === 'running' && s.kind === 'synthesizing')) {
   576→   576→                  steps = [...steps, {
   577→   577→                    id: newStepId(), kind: 'synthesizing',
   578→   578→                    label: 'Synthesizing response',
   579→   579→                    status: 'running', ts: Date.now(),
   580→   580→                  }];
   581→   581→                }
   582→   582→                // Update matching tool call with result
   583→   583→                const toolCalls = (m.toolCalls ?? []).map(tc => {
   584→   584→                  // Match by name AND status running (the most recent matching one)
   585→   585→                  if (tc.toolName === event.toolName && tc.status === 'running') {
   586→   586→                    return {
   587→   587→                      ...tc,
   588→   588→                      result: event.result,
   589→   589→                      durationMs: event.durationMs,
   590→   590→                      status: 'success' as const,
   591→   591→                    };
   592→   592→                  }
   593→   593→                  return tc;
   594→   594→                });
   595→   595→                return { ...m, thinkingSteps: steps, toolCalls };
   596→   596→              });
   597→   597→              break;
   598→   598→            }
   599→   599→            case 'handoff_request': {
   600→   600→              updateAssistant(m => ({
   601→   601→                ...m,
   602→   602→                thinkingSteps: [
   603→   603→                  ...(m.thinkingSteps ?? []),
   604→   604→                  {
   605→   605→                    id: newStepId(), kind: 'info',
   606→   606→                    label: `Handing off to ${event.target}`,
   607→   607→                    detail: event.reason,
   608→   608→                    status: 'completed', ts: Date.now(),
   609→   609→                  },
   610→   610→                ],
   611→   611→              }));
   612→   612→              break;
   613→   613→            }
   614→   614→            case 'subagent_spawned': {
   615→   615→              updateAssistant(m => ({
   616→   616→                ...m,
   617→   617→                thinkingSteps: [
   618→   618→                  ...(m.thinkingSteps ?? []),
   619→   619→                  {
   620→   620→                    id: newStepId(), kind: 'analyzing',
   621→   621→                    label: `Spawned ${event.agentType} subagent`,
   622→   622→                    status: 'completed', ts: Date.now(),
   623→   623→                  },
   624→   624→                ],
   625→   625→              }));
   626→   626→              break;
   627→   627→            }
   628→   628→            case 'completed': {
   629→   629→              const totalDuration = Date.now() - startedAt;
   630→   630→              updateAssistant(m => ({
   631→   631→                ...m,
   632→   632→                isStreaming: false,
   633→   633→                content: m.content || event.output?.content || accumulatedContent,
   634→   634→                thinkingSteps: finishRunningSteps(m.thinkingSteps ?? []),
   635→   635→                metadata: {
   636→   636→                  tokensUsed: event.tokensUsed || event.output?.metadata?.tokensUsed,
   637→   637→                  cost: event.cost ?? event.output?.metadata?.cost,
   638→   638→                  durationMs: event.output?.metadata?.durationMs ?? totalDuration,
   639→   639→                },
   640→   640→              }));
   641→   641→              break;
   642→   642→            }
   643→   643→            case 'error': {
   644→   644→              const errMsg = event.error?.message || 'Agent error';
   645→   645→              updateAssistant(m => ({
   646→   646→                ...m,
   647→   647→                isStreaming: false,
   648→   648→                isError: true,
   649→   649→                content: m.content
   650→   650→                  ? m.content + `\n\n⚠️ **Error:** ${errMsg}`
   651→   651→                  : `⚠️ **Error:** ${errMsg}`,
   652→   652→                thinkingSteps: m.thinkingSteps?.map(s =>
   653→   653→                  s.status === 'running'
   654→   654→                    ? { ...s, status: 'failed' as const }
   655→   655→                    : s
   656→   656→                ) ?? [],
   657→   657→              }));
   658→   658→              setError(errMsg);
   659→   659→              break;
   660→   660→            }
   661→   661→            case 'cancelled': {
   662→   662→              updateAssistant(m => ({
   663→   663→                ...m,
   664→   664→                isStreaming: false,
   665→   665→                content: m.content + '\n\n_(cancelled)_',
   666→   666→                thinkingSteps: m.thinkingSteps?.map(s =>
   667→   667→                  s.status === 'running'
   668→   668→                    ? { ...s, status: 'failed' as const, detail: 'cancelled' }
   669→   669→                    : s
   670→   670→                ) ?? [],
   671→   671→              }));
   672→   672→              break;
   673→   673→            }
   674→   674→            case 'message_saved': {
   675→   675→              // Replace temp id with real DB id
   676→   676→              setMessages(prev => prev.map(m =>
   677→   677→                m.id === assistantId ? { ...m, id: event.messageId } : m
   678→   678→              ));
   679→   679→              currentAssistantIdRef.current = event.messageId;
   680→   680→              // Refresh sessions to update lastActivityAt + counts
   681→   681→              refreshSessions();
   682→   682→              break;
   683→   683→            }
   684→   684→          }
   685→   685→        }
   686→   686→      }
   687→   687→
   688→   688→      // If no content arrived, show a fallback
   689→   689→      updateAssistant(m => {
   690→   690→        if (m.content || m.toolCalls?.length || m.artifacts?.length) return m;
   691→   691→        return { ...m, content: '_(No response received)_', isStreaming: false };
   692→   692→      });
   693→   693→    } catch (err: any) {
   694→   694→      if (err.name === 'AbortError') {
   695→   695→        updateAssistant(m => ({
   696→   696→          ...m,
   697→   697→          isStreaming: false,
   698→   698→          content: (m.content || '') + '\n\n_(stopped by user)_',
   699→   699→          thinkingSteps: m.thinkingSteps?.map(s =>
   700→   700→            s.status === 'running'
   701→   701→              ? { ...s, status: 'failed' as const, detail: 'stopped' }
   702→   702→              : s
   703→   703→          ) ?? [],
   704→   704→        }));
   705→   705→      } else {
   706→   706→        const msg = err.message || 'Network error';
   707→   707→        updateAssistant(m => ({
   708→   708→          ...m,
   709→   709→          isStreaming: false,
   710→   710→          isError: true,
   711→   711→          content: m.content
   712→   712→            ? m.content + `\n\n⚠️ **Error:** ${msg}`
   713→   713→            : `⚠️ **Error:** ${msg}`,
   714→   714→        }));
   715→   715→        setError(msg);
   716→   716→      }
   717→   717→    } finally {
   718→   718→      setSending(false);
   719→   719→      abortRef.current = null;
   720→   720→      currentAssistantIdRef.current = null;
   721→   721→    }
   722→   722→  }, [input, currentSessionId, sending, agents, router, refreshSessions]);
   723→   723→
   724→   724→  // ── Stop generation ────────────────────────────────────────────────────────
   725→   725→  const handleStop = useCallback(() => {
   726→   726→    abortRef.current?.abort();
   727→   727→    setSending(false);
   728→   728→  }, []);
   729→   729→
   730→   730→  // ── Regenerate last assistant message ──────────────────────────────────────
   731→   731→  const handleRegenerate = useCallback(() => {
   732→   732→    if (sending || messages.length < 2) return;
   733→   733→    // Find last user message
   734→   734→    let lastUserContent = '';
   735→   735→    for (let i = messages.length - 1; i >= 0; i--) {
   736→   736→      if (messages[i].role === 'user') {
   737→   737→        lastUserContent = messages[i].content;
   738→   738→        break;
   739→   739→      }
   740→   740→    }
   741→   741→    if (!lastUserContent) return;
   742→   742→    // Remove last assistant message
   743→   743→    setMessages(prev => {
   744→   744→      const next = [...prev];
   745→   745→      // remove trailing assistant messages
   746→   746→      while (next.length && next[next.length - 1].role === 'assistant') next.pop();
   747→   747→      return next;
   748→   748→    });
   749→   749→    // Re-send
   750→   750→    setTimeout(() => sendMessage(lastUserContent), 50);
   751→   751→  }, [sending, messages, sendMessage]);
   752→   752→
   753→   753→  // ── Textarea key handler ───────────────────────────────────────────────────
   754→   754→  const onKeyDown = (e: KeyboardEvent<HTMLTextAreaElement>) => {
   755→   755→    if (e.key === 'Enter' && !e.shiftKey) {
   756→   756→      e.preventDefault();
   757→   757→      sendMessage();
   758→   758→    }
   759→   759→  };
   760→   760→
   761→   761→  // ── Derived data ───────────────────────────────────────────────────────────
   762→   762→  const currentAgent = useMemo(
   763→   763→    () => agents.find(a => a.id === currentSessionMeta?.agentId),
   764→   764→    [agents, currentSessionMeta],
   765→   765→  );
   766→   766→
   767→   767→  // ── Loading state ──────────────────────────────────────────────────────────
   768→   768→  if (loadingData) {
   769→   769→    return (
   770→   770→      <div className="min-h-screen flex items-center justify-center bg-background">
   771→   771→        <div className="flex flex-col items-center gap-3">
   772→   772→          <RefreshCw className="w-8 h-8 animate-spin text-muted-foreground" />
   773→   773→          <p className="text-sm text-muted-foreground">Loading chat…</p>
   774→   774→        </div>
   775→   775→      </div>
   776→   776→    );
   777→   777→  }
   778→   778→
   779→   779→  // ── Render ─────────────────────────────────────────────────────────────────
   780→   780→  return (
   781→   781→    <TooltipProvider delayDuration={200}>
   782→   782→      <div className="h-screen w-full flex bg-background overflow-hidden">

   784→   784→        <ChatSidebar

   799→   799→        <main className="flex-1 flex flex-col min-w-0 bg-gradient-to-b from-background to-slate-50/50 dark:to-slate-950/30">
   800→   800→          {/* Header */}
   801→   801→          <header className="flex-shrink-0 border-b bg-background/80 backdrop-blur-sm">

   804→   804→              <Button

   920→   920→          <div
